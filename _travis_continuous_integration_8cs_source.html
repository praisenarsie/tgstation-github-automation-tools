124  public async Task TriggerJobRestart(PullRequest pullRequest, CancellationToken cancellationToken)
125  {

129  var statuses = await gitHubManager.GetLatestCommitStatus(pullRequest).ConfigureAwait(false);
130  var buildNumberRegex = new Regex(@"/builds/([1-9][0-9]*)\?");
131  var tasks = new List<Task>();
132  foreach (var I in statuses.Statuses)

137
